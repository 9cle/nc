// --- THE 24 KEYWORDS OF N ---

// 1-5: Structure
import   // Load modules
std      // Standard namespace
void     // Define a function/no return
pub      // Public access
type     // Custom types

// 6-9: Data
let      // Constant
var      // Variable
mut      // Mutable reference
nil      // Null value

// 10-15: Logic
if       // Condition
else     // Fallback
match    // Pattern matching
loop     // To repeat code
break    // Exit
continue // Skip

// 16-19: Values & Types
int      // Integer type
true     // Truth
false    // False
is       // Type check

// 20-24: Systems
as       // Casting
defer    // Cleanup at end
test     // Unit testing
error    // Error handling
return   // Exit function
